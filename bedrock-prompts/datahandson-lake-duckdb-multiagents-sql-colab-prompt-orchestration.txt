Here are examples of DuckDB queries <duckdb_examples>.

<duckdb_examples>

<duckdb_example>
SELECT 
    genre,
    total_movies,
    avg_rating,
    rank_by_quality
FROM s3_tables_db.gold.gold_genre_analytics
ORDER BY avg_rating DESC
LIMIT 10;
</duckdb_example>

<duckdb_example>
SELECT 
    movie_title,
    similar_movie_title,
    recommendation_score
FROM s3_tables_db.gold.gold_movie_recommendations
WHERE movie_title = 'Toy Story'
ORDER BY recommendation_score DESC
LIMIT 5;
</duckdb_example>

<duckdb_example>
SELECT
    year,
    SUM(total_ratings) AS total_ratings,
    AVG(avg_rating) AS avg_rating
FROM s3_tables_db.gold.gold_temporal_trends
GROUP BY year
ORDER BY year;
</duckdb_example>

<duckdb_example>
SELECT
    favorite_genre,
    COUNT(*) AS users
FROM s3_tables_db.gold.gold_user_behavior
GROUP BY favorite_genre
ORDER BY users DESC;
</duckdb_example>

</duckdb_examples>
